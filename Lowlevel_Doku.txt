    ir_lib.read_calibrated_values(Values);

        IR_Values[i] = Values[i] - (ir_lib.offset_save[i]);

// Line.cpp
// Liest Linienwerte von ADC-Expander
void LineC::read(){
    Summe = 0;
    for(int y = 0 ; y<4 ; y++){
        for(int i = 0 ; i<8 ; i++){
            if(((y*8)+i) == 13){ // pt 13 kaputt
                Raw[13] = Expander.ADC.give_digital(list[1],4);
                Summe = Summe + Raw[13];
            }
            else{
                Raw[(y*8)+i] = Expander.ADC.give_digital(list[y],i);
                Summe = Summe + Raw[(y*8)+i];
            }
        }
    }
}

// Liest VW-Linienwerte
void LineC::read_VW(){
    Summe_VW = 0;
    for(int i = 0 ; i<8 ; i++){
        Raw_VW[i] = Expander.ADC.give_digital(CS_LineVW,i);
        Summe_VW = Summe_VW + Raw_VW[i];
    }
}

void MouseC::read(){
    PMW3389_DATA mouseData = pmw.readBurst();
    deltaX = -1 * mouseData.dx * 1.25 / 100;
    deltaY = mouseData.dy * 1.25 / 100;
    delta_dist = sqrtf(pow(deltaY,2)+pow(deltaX,2));
    xPos = xPos + deltaX ;
    yPos = yPos + deltaX ;
    lift = mouseData.isOnSurface;
    movement = mouseData.isMotion;
}

// PIDCalculations.cpp
// Berechnet PID-Regelung
void PIDC::Calculate(){
    now = micros();
    diffTime = double(now - last) / 1000;
    last = now;
    double error = Angle - BNO055.give_TiltZ();
    esum = esum + double(error);
    const int max_error_sum = 6000;
    if(esum > 0)
        esum = min(esum, max_error_sum);
    else
        esum = max(esum, -max_error_sum);
    if(Utils.isinRange(error, -1, 1)) {
        esum = 0;
    }
    double derivative = (error - ealt) / diffTime;
    ealt = error;
    if (error > 0){
        Out  = (error * Kp  * PID_Mult) + (esum * Ki * diffTime * PID_Mult) + (derivative * Kd);
    }else{
        Out  = (error * Kp  * PID_Mult) + -(esum * Ki * diffTime * PID_Mult) + -(derivative * Kd);
    }
    last = micros(); 
}

    Angle = Angle1;
}

// Schreibt Farbe auf LED
void RGBC::write(int Pos,String color){
    if (color == "R"){
        R[Pos]=255;
        G[Pos]=0;
        B[Pos]=0;
    }
    else if (color == "G"){
        R[Pos]=0;
        G[Pos]=255;
        B[Pos]=0;
    }
    else if (color == "B"){
        R[Pos]=0;
        G[Pos]=0;
        B[Pos]=255;
    }
    else if (color == "O"){
        R[Pos]=255;
        G[Pos]=70;
        B[Pos]=0;
    }
    else if (color == "Y"){
        R[Pos]=255;
        G[Pos]=255;
        B[Pos]=30;
    }
    else{
        R[Pos]=0;
        G[Pos]=0;
        B[Pos]=0;
    }
}
